# PROMPT — INSTALLATION MODULE 05 — BRAND SOUND

```
Ajoute le MODULE 05 — BRAND SOUND au projet existant.

Le Module 05 génère des prompts optimisés pour RoboNeo.com et ElevenLabs afin de créer l'identité sonore d'une marque: jingle (5-10s), musiques de fond (15/30/60s), beat detection pour synchronisation, vocal separation, UGC audio enhancement, et sound effects.

Le module doit s'intégrer avec les Modules 01, 02, 03 et 04 déjà installés.

Ajoute ce dossier et ces fichiers:

```

├── __init__.py
├── orchestrator.py
└── prompts.py
```

def get_jingle_prompt(sector: str, brand_name: str) -> str:
    """Génère un prompt pour le jingle"""
    prompt = JINGLE_TEMPLATES.get(sector, JINGLE_TEMPLATES["luxe"])
    return f"{prompt}\n\nNom de la marque: {brand_name}"

def get_background_music_prompt(sector: str, duration: str) -> str:
    """Génère un prompt pour la musique de fond"""
    sector_templates = BACKGROUND_MUSIC_TEMPLATES.get(sector, BACKGROUND_MUSIC_TEMPLATES["luxe"])
    return sector_templates.get(duration, sector_templates["30s"])

def get_sound_effect_prompt(effect_name: str) -> str:
    """Génère un prompt pour un effet sonore"""
    return SOUND_EFFECTS.get(effect_name, SOUND_EFFECTS["ui_click"])

def get_voice_over_voice(sector: str) -> dict:
    """Retourne la voix recommandée pour un secteur"""
    for voice_id, voice in VOICE_OVER_VOICES.items():
        if sector in voice["use_for"]:
            return voice
    return VOICE_OVER_VOICES["professionnel"]

def get_audio_processing_prompt(process_type: str) -> str:
    """Génère un prompt pour le traitement audio"""
    return AUDIO_PROCESSING_PROMPTS.get(process_type, AUDIO_PROCESSING_PROMPTS["denoise"])

from typing import Dict, Any, List
from modules.base_module import BaseModule
from modules.module_05_brand_sound.prompts import (
    get_jingle_prompt,
    get_background_music_prompt,
    get_sound_effect_prompt,
    get_voice_over_voice,
    get_audio_processing_prompt
)

class BrandSoundOrchestrator(BaseModule):
    """
    MODULE 05 — BRAND SOUND
    Génère tous les prompts pour créer l'identité sonore d'une marque
    """
    
    def __init__(self):
        super().__init__("BrandSound")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 05
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "tone": "luxe",
            "values": ["excellence", "prestige"],
            "target_audience": "femmes_35_50",
            "has_ugc_audio": False,  # si l'utilisateur a des audios UGC à nettoyer
            "needs_vocal_separation": False
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        tone = brief.get("tone", "professionnel")
        values = brief.get("values", ["qualité", "confiance"])
        target_audience = brief.get("target_audience", "mixte")
        has_ugc_audio = brief.get("has_ugc_audio", False)
        needs_vocal_separation = brief.get("needs_vocal_separation", False)
        
        self._log(f"Génération des prompts sonores pour {brand_name} ({sector})")
        
        # 1. Jingle (5-10s)
        jingle_prompt = get_jingle_prompt(sector, brand_name)
        jingle_enhanced = self._enhance_prompt(jingle_prompt, {
            "brand": brand_name,
            "sector": sector,
            "tone": tone
        })
        
        jingle = {
            "prompt": jingle_enhanced,
            "parameters": {
                "duration": "5-10s",
                "format": ["MP3", "WAV"],
                "use_for": ["intros vidéo", "notifications", "podcasts"]
            }
        }
        
        # 2. Background Music (15s, 30s, 60s)
        bgm = {}
        durations = ["15s", "30s", "60s"]
        
        for duration in durations:
            prompt = get_background_music_prompt(sector, duration)
            personalized = self._personalize_prompt(prompt, {
                "brand": brand_name,
                "sector": sector
            })
            enhanced = self._enhance_prompt(personalized, {
                "brand": brand_name,
                "duration": duration
            })
            
            bgm[duration] = {
                "prompt": enhanced,
                "parameters": {
                    "duration": duration,
                    "format": ["MP3", "WAV"],
                    "loopable": True,
                    "use_for": self._get_bgm_use_for(duration)
                }
            }
        
        # 3. Sound Effects Library
        sound_effects = {}
        effect_names = ["ui_click", "ui_notification", "ui_success", "transition_whoosh", "transition_sweep", "impact_soft"]
        
        for effect_name in effect_names:
            prompt = get_sound_effect_prompt(effect_name)
            enhanced = self._enhance_prompt(prompt, {"brand": brand_name})
            
            sound_effects[effect_name] = {
                "prompt": enhanced,
                "parameters": {
                    "duration": self._get_effect_duration(effect_name),
                    "format": "MP3",
                    "use_for": self._get_effect_use_for(effect_name)
                }
            }
        
        # 4. Voice-Over Recommendation
        voice = get_voice_over_voice(sector)
        
        voice_over = {
            "recommended_voice": {
                "id": voice["id"],
                "name": voice["name"],
                "description": voice["description"],
                "tone": voice["tone"]
            },
            "parameters": {
                "stability": voice["stability"],
                "similarity_boost": voice["similarity"],
                "format": "MP3",
                "sample_rate": 44100
            },
            "prompt_template": f"Utilisez la voix {voice['name']} sur ElevenLabs. Ton: {voice['tone']}. Script fourni dans les modules 03 (Video Content) et 06 (Copy & Content)."
        }
        
        # 5. Audio Processing (si UGC à nettoyer)
        audio_processing = {}
        
        if has_ugc_audio:
            audio_processing["denoise"] = {
                "prompt": get_audio_processing_prompt("denoise"),
                "parameters": {
                    "input": "audio UGC fourni par le client",
                    "output_format": "MP3",
                    "target_lufs": -14
                }
            }
        
        if needs_vocal_separation:
            audio_processing["vocal_separation"] = {
                "prompt": get_audio_processing_prompt("vocal_separation"),
                "parameters": {
                    "input": "audio source",
                    "output_stems": ["vocal", "instrumental", "bass", "drums", "other"],
                    "format": "MP3"
                }
            }
        
        # 6. Beat Detection & Sync (pour synchronisation vidéo)
        beat_detection = {
            "prompt": """
Analyse l'audio fourni et génère:
- BPM détecté
- Position des beats (timestamps)
- Downbeats (premiers temps des mesures)
- Points de coupe recommandés pour montage vidéo
- Format sortie: JSON avec timestamps
""",
            "parameters": {
                "input": "fichier audio (MP3/WAV)",
                "output": "beat_timestamps.json",
                "use_for": "synchronisation vidéo avec Module 03"
            }
        }
        
        # Retourner tous les prompts
        return {
            "brand_sound": {
                "jingle": {
                    "agent": "AI Music Generator / AI Background Music Generator",
                    "prompts": jingle,
                    "parameters": {
                        "sector": sector,
                        "duration": "5-10s",
                        "brand_name_integration": True
                    }
                },
                "background_music": {
                    "agent": "AI Background Music Generator",
                    "prompts": bgm,
                    "parameters": {
                        "durations": ["15s", "30s", "60s"],
                        "sector": sector,
                        "loopable": True
                    }
                },
                "sound_effects": {
                    "agent": "AI Music Generator / Sound Library",
                    "effects": sound_effects,
                    "parameters": {
                        "effects_count": len(effect_names),
                        "formats": ["MP3"],
                        "categories": ["UI", "transitions", "impacts"]
                    }
                },
                "voice_over": {
                    "agent": "ElevenLabs / AI Voice Generator",
                    "recommendation": voice_over,
                    "available_voices": [
                        {"name": v["name"], "description": v["description"], "tone": v["tone"]}
                        for v in [get_voice_over_voice(s) for s in ["chaleureux", "professionnel", "dynamique", "élégant", "naturel"]]
                    ]
                },
                "audio_processing": audio_processing if (has_ugc_audio or needs_vocal_separation) else None,
                "beat_detection": {
                    "agent": "Audio Analysis (librosa)",
                    "prompt": beat_detection["prompt"],
                    "parameters": beat_detection["parameters"]
                }
            },
            "metadata": {

    def _personalize_prompt(self, prompt: str, context: Dict) -> str:

        for key, value in context.items():

        return prompt
