```
Ajoute le MODULE 09 — UPSELL & CROSS-SELL KIT au projet existant.

def get_cross_sell_ideas(sector: str) -> list:
    """Retourne 3 idées de produits complémentaires"""
    return CROSS_SELL_IDEAS.get(sector, CROSS_SELL_IDEAS["bijou"])

def get_bundle_offer(bundle_type: str) -> dict:
    """Retourne une offre groupée"""
    return BUNDLE_OFFERS.get(bundle_type, BUNDLE_OFFERS["standard"])

def get_bundle_copy(copy_type: str, context: dict) -> dict:
    """Génère une copy pour offre groupée"""
    template = BUNDLE_COPY_TEMPLATES.get(copy_type, BUNDLE_COPY_TEMPLATES["upsell_checkout"])
    copy = {}
    for key, value in template.items():
        text = value
        for k, v in context.items():
            text = text.replace(f"{{{k}}}", str(v))
        copy[key] = text
    return copy

## 3. FICHIER modules/module_09_upsell_kit/orchestrator.py

```python
from typing import Dict, Any, List
from modules.base_module import BaseModule
from modules.module_09_upsell_kit.prompts import (
    get_cross_sell_ideas,
    get_bundle_offer,
    get_bundle_copy
)

class UpsellKitOrchestrator(BaseModule):
    """
    MODULE 09 — UPSELL & CROSS-SELL KIT
    Génère les stratégies et contenus pour maximiser la valeur du panier
    """
    
    def __init__(self):
        super().__init__("UpsellKit")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les éléments du Module 09
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "product_name": "Montre Élégance Or Rose",
            "product_price": 299,
            "main_product_features": ["or rose", "verre saphir", "mouvement suisse"]
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_name = brief.get("product_name", "notre produit")
        product_price = brief.get("product_price", 299)
        main_features = brief.get("main_product_features", [])
        
        self._log(f"Génération des upsell/cross-sell pour {brand_name}")
        
        # 1. Idées de produits complémentaires (3)
        cross_sell_ideas = get_cross_sell_ideas(sector)
        
        # Personnaliser les idées avec le contexte de la marque
        personalized_ideas = []
        for idea in cross_sell_ideas[:3]:
            personalized = idea.copy()
            personalized["product_name"] = personalized["product_name"]
            
            # Ajouter la justification personnalisée
            if sector == "bijou" and "montre" in product_name.lower():
                personalized["justification"] = f"Complète parfaitement {product_name}. Idéal pour protéger votre investissement."
            
            personalized_ideas.append(personalized)
        
        # 2. Visuels bundle (prompts pour générer les visuels)
        bundle_visuals = []
        for idea in personalized_ideas:
            visual_prompt = idea.get("visual_prompt", "")
            
            # Personnaliser le prompt avec le produit principal
            custom_prompt = f"""
{visual_prompt}

Style: Cohérent avec l'identité de la marque {brand_name}.
Association avec le produit principal: {product_name}.
"""
            bundle_visuals.append({
                "product": idea["product_name"],
                "prompt": custom_prompt,
                "agent": "Image Generation / AI Poster Agent",
                "parameters": {
                    "dimensions": "1080x1080",
                    "format": "jpg/png",
                    "style": "cohérent avec la marque"
                }
            })
        
        # 3. Bundle offers (3 types)
        bundle_offers = []
        bundle_types = ["standard", "premium", "gift"]
        
        for i, bundle_type in enumerate(bundle_types):
            bundle = get_bundle_offer(bundle_type)
            
            # Prix du bundle
            if i < len(personalized_ideas):
                accessory_price = personalized_ideas[i].get("price_range", "29-49€").split("-")[0].replace("€", "")
                try:
                    accessory_price_num = int(accessory_price)
                except:
                    accessory_price_num = 39
                
                total_price = product_price + accessory_price_num
                discounted_price = total_price * (1 - bundle["discount"] / 100)
                
                bundle_context = {
                    "product": product_name,
                    "accessory": personalized_ideas[i]["product_name"],
                    "accessory_price": accessory_price_num,
                    "total_price": total_price,
                    "discounted_price": round(discounted_price, 2),
                    "discount": bundle["discount"],
                    "saving": round(total_price - discounted_price, 2)
                }
            else:
                bundle_context = {
                    "product": product_name,
                    "accessory": "nos accessoires",
                    "accessory_price": 0,
                    "total_price": product_price,
                    "discounted_price": product_price,
                    "discount": bundle["discount"],
                    "saving": 0
                }
            
            # Copy pour le bundle
            bundle_copy = get_bundle_copy("bundle_page", bundle_context)
            
            bundle_offers.append({
                "name": bundle["name"],
                "description": bundle["description"],
                "discount": bundle["discount"],
                "cta": bundle["cta"],
                "products": [product_name] + [personalized_ideas[i]["product_name"] for i in range(min(len(personalized_ideas), 2))],
                "original_price": bundle_context["total_price"],
                "bundle_price": bundle_context["discounted_price"],
                "saving": bundle_context["saving"],
                "copy": bundle_copy,
                "visual_prompt": bundle["visual_prompt"]
            })
        
        # 4. Copy pour upsell/cross-sell (tous contextes)
        upsell_copy = {}
        
        # Upsell page produit
        if personalized_ideas:
            upsell_copy["product_page"] = get_bundle_copy("upsell_checkout", {
                "accessory": personalized_ideas[0]["product_name"],
                "price": personalized_ideas[0].get("price_range", "29-49€").split("-")[0],
                "original_price": personalized_ideas[0].get("price_range", "29-49€").split("-")[1] if "-" in personalized_ideas[0].get("price_range", "") else "49",
                "benefit_reason": personalized_ideas[0].get("justification", "parfait pour compléter votre achat")[:50]
            })
        
        # Cross-sell panier
        cross_sell_copies = []
        for idea in personalized_ideas:
            cross_sell_copies.append(get_bundle_copy("cross_sell_cart", {
                "product": product_name,
                "accessory": idea["product_name"],
                "accessory_benefit": idea.get("justification", "accessoire indispensable")[:60]
            }))
        upsell_copy["cart_page"] = cross_sell_copies
        
        # Post-achat
        upsell_copy["post_purchase"] = get_bundle_copy("post_purchase", {
            "discount": 15,
            "accessory": personalized_ideas[0]["product_name"] if personalized_ideas else "nos accessoires"
        })
        
        # Retourner tous les éléments
        return {
            "upsell_kit": {
                "cross_sell_ideas": {
                    "agent": "Manual (strategy)",
                    "ideas": personalized_ideas,
                    "parameters": {
                        "count": len(personalized_ideas),
                        "sector": sector,
                        "bundle_ready": True
                    }
                },
                "bundle_visuals": {
                    "agent": "Image Generation / AI Poster Agent",
                    "prompts": bundle_visuals,
                    "parameters": {
                        "dimensions": "1080x1080",
                        "format": "jpg/png",
                        "quantity": len(bundle_visuals)
                    }
                },
                "bundle_offers": {
                    "agent": "Manual (strategy)",
                    "offers": bundle_offers,
                    "parameters": {
                        "types": bundle_types,
                        "discount_range": "10-20%",
                        "max_products": 3
                    }
                },
                "upsell_copy": {
                    "agent": "Manual (copywriting)",
                    "copy": upsell_copy,
                    "parameters": {
                        "contexts": ["product_page", "cart_page", "post_purchase"],
                        "variants": len(upsell_copy)
                    }
                }
            },
            "metadata": {
                "module": "09_upsell_kit",
                "brand_name": brand_name,
                "sector": sector,
                "cross_sell_ideas": len(personalized_ideas),
                "bundle_offers": len(bundle_offers),
                "llm_enhanced": bool(self._enhance_prompt("test", {}) != "test")
            }
        }
```
