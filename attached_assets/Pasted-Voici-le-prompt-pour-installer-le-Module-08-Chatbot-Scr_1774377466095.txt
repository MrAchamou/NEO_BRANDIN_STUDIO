# PROMPT — INSTALLATION MODULE 08 — CHATBOT SCRIPT

```
Ajoute le MODULE 08 — CHATBOT SCRIPT au projet existant.

Le Module 08 génère les scripts pour le service client automatisé: 20 questions/réponses fréquentes sur le produit, script de gestion des objections (trop cher, délai livraison, retours), et réponses aux commentaires négatifs sur les réseaux sociaux.

Le module doit s'intégrer avec les Modules 01 à 07 déjà installés.

```

def get_faq(sector: str, context: dict) -> list:
    """Génère une FAQ de 20 questions/réponses"""
    templates = FAQ_TEMPLATES.get(sector, FAQ_TEMPLATES["bijou"])
    faq = []
    
    for qa in templates:
        question = qa["question"]
        answer = qa["answer"]
        for key, value in context.items():
            question = question.replace(f"{{{key}}}", str(value))
            answer = answer.replace(f"{{{key}}}", str(value))
        faq.append({"question": question, "answer": answer})
    
    # Compléter jusqu'à 20 questions si nécessaire
    while len(faq) < 20:
        faq.append({
            "question": "Puis-je contacter le service client ?",
            "answer": f"Oui, notre service client est joignable par email à {context.get('support_email', 'support@exemple.com')} ou par chat sur notre site. Nous répondons sous 24h ouvrées."
        })
    
    return faq[:20]

def get_objection_script(objection_type: str, context: dict) -> dict:
    """Génère un script de gestion d'objection"""
    template = OBJECTION_SCRIPTS.get(objection_type, OBJECTION_SCRIPTS["prix_trop_eleve"])
    response = template["response"]
    follow_up = template["follow_up"]
    
    for key, value in context.items():
        response = response.replace(f"{{{key}}}", str(value))
        follow_up = follow_up.replace(f"{{{key}}}", str(value))
    
    return {
        "scenario": template["scenario"],
        "response": response,
        "follow_up": follow_up
    }

def get_negative_comment_response(comment_type: str, context: dict) -> dict:
    """Génère une réponse à un commentaire négatif"""
    template = NEGATIVE_COMMENTS_TEMPLATES.get(comment_type, NEGATIVE_COMMENTS_TEMPLATES["produit_defectueux"])
    public = template["response_public"]
    private = template["response_prive"]
    
    for key, value in context.items():
        public = public.replace(f"{{{key}}}", str(value))
        private = private.replace(f"{{{key}}}", str(value))
    
    return {
        "type": template["type"],
        "public_response": public,
        "private_message": private
    }

from typing import Dict, Any, List
from modules.base_module import BaseModule
from modules.module_08_chatbot_script.prompts import (
    get_faq,
    get_objection_script,
    get_negative_comment_response
)

class ChatbotScriptOrchestrator(BaseModule):
    """
    MODULE 08 — CHATBOT SCRIPT
    Génère tous les scripts pour le service client automatisé
    """
    
    def __init__(self):
        super().__init__("ChatbotScript")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les scripts du Module 08
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "product_name": "Montre Élégance Or Rose",
            "product_description": "...",
            "material": "or rose 18 carats",
            "warranty": 2,
            "delivery_days": 3,
            "return_days": 30,
            "discount": 20,
            "promo_code": "LUXEOR20",
            "price": 299,
            "free_shipping": 100,
            "support_email": "contact@luxeor.com",
            "unique_feature": "mouvement suisse",
            "best_seller_1": "Collier Perle",
            "best_seller_2": "Bague Solitaire"
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_name = brief.get("product_name", "notre produit")
        material = brief.get("material", "matériaux d'exception")
        warranty = brief.get("warranty", 2)
        delivery_days = brief.get("delivery_days", 3)
        express_delivery_days = brief.get("express_delivery_days", 1)
        express_price = brief.get("express_price", 9.90)
        return_days = brief.get("return_days", 30)
        discount = brief.get("discount", 20)
        promo_code = brief.get("promo_code", brand_name.upper()[:4])
        price = brief.get("price", 299)
        discounted_price = round(price * (1 - discount / 100), 2)
        free_shipping = brief.get("free_shipping", 100)
        support_email = brief.get("support_email", f"contact@{brand_name.lower()}.com")
        unique_feature = brief.get("unique_feature", "fabrication artisanale")
        competitor_warranty = brief.get("competitor_warranty", 1)
        offer_end_days = brief.get("offer_end_days", 3)
        best_seller_1 = brief.get("best_seller_1", "notre best-seller")
        best_seller_2 = brief.get("best_seller_2", "notre coup de cœur")
        reason_1 = brief.get("reason_1", "il est très polyvalent")
        
        # Contexte pour personnalisation
        context = {
            "product": product_name,
            "brand": brand_name,
            "material": material,
            "warranty": warranty,
            "delivery_days": delivery_days,
            "express_delivery_days": express_delivery_days,
            "express_price": express_price,
            "return_days": return_days,
            "discount": discount,
            "code": promo_code,
            "price": price,
            "discounted_price": discounted_price,
            "free_shipping": free_shipping,
            "support_email": support_email,
            "unique_feature": unique_feature,
            "competitor_warranty": competitor_warranty,
            "offer_end_days": offer_end_days,
            "best_seller_1": best_seller_1,
            "best_seller_2": best_seller_2,
            "reason_1": reason_1
        }
        
        self._log(f"Génération des scripts chatbot pour {brand_name}")
        
        # 1. FAQ (20 questions/réponses)
        faq = get_faq(sector, context)
        
        # 2. Scripts de gestion des objections
        objections = {}
        objection_types = ["prix_trop_eleve", "delai_livraison", "hesitation_qualite", 
                          "comparaison_concurrent", "pas_besoin_maintenant", "frais_port", 
                          "taille_incertaine", "cadeau"]
        
        for obj_type in objection_types:
            objections[obj_type] = get_objection_script(obj_type, context)
        
        # 3. Réponses aux commentaires négatifs
        negative_responses = {}
        negative_types = ["produit_defectueux", "livraison_retard", "service_client", 
                         "produit_different", "prix_trop_eleve_apres_achat"]
        
        for neg_type in negative_types:
            negative_responses[neg_type] = get_negative_comment_response(neg_type, context)
        
        # Retourner tous les scripts
        return {
            "chatbot_script": {
                "faq": {
                    "agent": "Manual (chatbot configuration)",
                    "questions": faq,
                    "parameters": {
                        "total_questions": len(faq),
                        "sector": sector,
                        "format": "JSON pour intégration chatbot"
                    }
                },
                "objection_handling": {
                    "agent": "Manual (customer service training)",
                    "scripts": objections,
                    "parameters": {
                        "scenarios": len(objections),
                        "format": "scripts pour service client"
                    }
                },
                "negative_comments": {
                    "agent": "Manual (social media management)",
                    "responses": negative_responses,
                    "parameters": {
                        "scenarios": len(negative_responses),
                        "format": "public response + private message"
                    }
                }
            },
            "metadata": {
                "module": "08_chatbot_script",
                "brand_name": brand_name,
                "sector": sector,
                "faq_questions": len(faq),
                "objection_scenarios": len(objections),
                "negative_scenarios": len(negative_responses),
                "llm_enhanced": bool(self._enhance_prompt("test", {}) != "test")
            }
        }
