def get_dashboard_template(brand_name: str, context: dict) -> str:
    """Génère le template du tableau de bord Google Sheets"""
    dashboard = DASHBOARD_TEMPLATE
    for key, value in context.items():
        dashboard = dashboard.replace(f"{{{key}}}", str(value))
    dashboard = dashboard.replace("{brand_name}", brand_name)
    return dashboard

def get_kpi_guide(context: dict) -> str:
    """Génère le guide des KPIs"""
    guide = KPI_GUIDE
    for key, value in context.items():
        guide = guide.replace(f"{{{key}}}", str(value))
    return guide

def get_scaling_guide(context: dict) -> str:
    """Génère le guide de scaling"""
    guide = SCALING_GUIDE
    for key, value in context.items():
        guide = guide.replace(f"{{{key}}}", str(value))
    return guide

def get_weekly_review_template(week_number: int, context: dict) -> str:
    """Génère le template d'analyse hebdomadaire"""
    template = WEEKLY_REVIEW_TEMPLATE
    template = template.replace("{week_number}", str(week_number))
    template = template.replace("{next_week_number}", str(week_number + 1))
    for key, value in context.items():
        template = template.replace(f"{{{key}}}", str(value))
    return template
