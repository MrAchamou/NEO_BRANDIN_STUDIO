def get_script_template(sector: str, duration: str) -> dict:
    """Retourne le template de script pour un secteur et une durée"""
    sector_templates = SCRIPT_TEMPLATES.get(sector, SCRIPT_TEMPLATES["bijou"])
    return sector_templates.get(duration, sector_templates["30s"])

def get_teaser_style_prompt(style: str) -> str:
    """Retourne le prompt pour un style de teaser"""
    return TEASER_STYLES.get(style, TEASER_STYLES["cinematic"])

def get_thumbnail_prompt(thumbnail_type: str) -> str:
    """Retourne le prompt pour un type de miniature"""
    return THUMBNAIL_PROMPTS.get(thumbnail_type, THUMBNAIL_PROMPTS["product_focus"])

def get_voice_over_voice(sector: str) -> dict:
    """Retourne la voix off recommandée pour un secteur"""
    for voice_id, voice in VOICE_OVER_VOICES.items():
        if sector in voice["use_for"]:
            return voice
    return VOICE_OVER_VOICES["professionnel"]

from typing import Dict, Any
from modules.base_module import BaseModule
from modules.module_03_video_content.prompts import (
    get_script_template,
    get_teaser_style_prompt,
    get_thumbnail_prompt,
    get_voice_over_voice
)

class VideoContentOrchestrator(BaseModule):
    """
    MODULE 03 — VIDEO CONTENT
    Génère tous les prompts pour créer le contenu vidéo d'une marque
    """
    
    def __init__(self):
        super().__init__("VideoContent")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 03
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "product_name": "Montre Élégance",
            "product_description": "Montre en or rose...",
            "product_features": ["or rose", "verre saphir", "mouvement automatique"],
            "benefits": ["élégance", "précision", "héritage"],
            "target_audience": "femmes_35_50",
            "visual_assets": {}  # du Module 02 (optionnel, pour contexte)
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_name = brief.get("product_name", "notre produit")
        product_description = brief.get("product_description", "")
        product_features = brief.get("product_features", [])
        benefits = brief.get("benefits", [])
        target_audience = brief.get("target_audience", "mixte")
        
        # Variables pour personnalisation des scripts
        year = brief.get("year", "1923")
        material = product_features[0] if product_features else "matériaux d'exception"
        detail = product_features[1] if len(product_features) > 1 else "finitions soignées"
        code = brief.get("promo_code", brand_name.upper()[:4])
        duration_days = brief.get("duration_days", "7")
        
        # Déterminer le thumbnail_type selon le secteur
        thumbnail_type = self._get_thumbnail_type(sector)
        
        # Déterminer le teaser_style selon le secteur
        teaser_style = self._get_teaser_style(sector)
        
        # Déterminer la voix off recommandée
        voice = get_voice_over_voice(sector)
        
        self._log(f"Génération des prompts vidéo pour {brand_name} ({sector})")
        
        # 1. Scripts pour chaque format (15s, 30s, 60s)
        scripts = {}
        durations = ["15s", "30s", "60s"]
        
        for duration in durations:
            template = get_script_template(sector, duration)
            
            # Personnaliser chaque section du script
            personalized = {}
            for section, text in template.items():
                personalized[section] = self._personalize_script(
                    text,
                    brand_name=brand_name,
                    product_name=product_name,
                    material=material,
                    detail=detail,
                    year=year,
                    code=code,
                    duration_days=duration_days,
                    benefits=benefits,
                    target_audience=target_audience
                )
            
            # Assembler le script complet
            full_script = " ".join(personalized.values())
            
            scripts[duration] = {
                "sections": personalized,
                "full_script": full_script,
                "voice_over_text": full_script
            }
        
        # 2. Prompts vidéo par format
        video_prompts = {
            "tiktok_15s": self._generate_video_prompt(
                duration="15s",
                format="vertical",
                product_name=product_name,
                sector=sector,
                script=scripts["15s"]["full_script"]
            ),
            "tiktok_30s": self._generate_video_prompt(
                duration="30s",
                format="vertical",
                product_name=product_name,
                sector=sector,
                script=scripts["30s"]["full_script"]
            ),
            "youtube_60s": self._generate_video_prompt(
                duration="60s",
                format="horizontal",
                product_name=product_name,
                sector=sector,
                script=scripts["60s"]["full_script"]
            )
        }
        
        # 3. Teaser animé
        teaser_prompt = get_teaser_style_prompt(teaser_style)
        teaser_prompt_customized = self._customize_teaser_prompt(
            teaser_prompt, product_name, brand_name, sector
        )
        
        # 4. Miniatures
        thumbnail_prompt = get_thumbnail_prompt(thumbnail_type)
        thumbnail_prompt_customized = self._customize_thumbnail_prompt(
            thumbnail_prompt, product_name, brand_name
        )
        
        # 5. Voix off (recommandation)
        voice_over = {
            "recommended_voice": voice["id"],
            "description": voice["description"],
            "tone": voice["tone"],
            "voice_prompt": f"Utilisez la voix {voice['id']} sur ElevenLabs avec le script fourni. Ton: {voice['tone']}."
        }
        
        # Retourner tous les prompts
        return {
            "video_content": {
                "scripts": {
                    "agent": "Creative Ad Video Agent / Manual (pour scripts)",
                    "prompts": scripts,
                    "parameters": {
                        "formats": ["15s", "30s", "60s"],
                        "sector": sector,
                        "personalized": True
                    }
                },
                "short_videos": {
                    "agent": "Creative Ad Video Agent",
                    "prompts": {
                        "tiktok_15s": video_prompts["tiktok_15s"],
                        "tiktok_30s": video_prompts["tiktok_30s"]
                    },
                    "parameters": {
                        "format": "9:16 (1080x1920)",
                        "duration": ["15s", "30s"],
                        "music": "à synchroniser avec Module 05"
                    }
                },
                "long_video": {
                    "agent": "Creative Ad Video Agent",
                    "prompts": {
                        "youtube_60s": video_prompts["youtube_60s"]
                    },
                    "parameters": {
                        "format": "16:9 (1920x1080)",
                        "duration": "60s",
                        "thumbnail": "à générer séparément"
                    }
                },
                "teaser": {
                    "agent": "Anime Video Agent / Image-to-Video",
                    "prompt": teaser_prompt_customized,
                    "parameters": {
                        "style": teaser_style,
                        "duration": "15s",
                        "format": "9:16 et 16:9",
                        "effects": ["selon style sélectionné"]
                    }
                },
                "thumbnails": {
                    "agent": "AI Poster Agent",
                    "prompt": thumbnail_prompt_customized,
                    "parameters": {
                        "type": thumbnail_type,
                        "dimensions": "1280x720",
                        "format": "jpg"
                    }
                },
                "voice_over": {
                    "agent": "ElevenLabs / AI Voice Generator",
                    "recommendation": voice_over,
                    "parameters": {
                        "voices_available": list(VOICE_OVER_VOICES.keys()),
                        "scripts": {
                            "15s": scripts["15s"]["voice_over_text"],
                            "30s": scripts["30s"]["voice_over_text"],
                            "60s": scripts["60s"]["voice_over_text"]
                        }
                    }
                }
            },
            "metadata": {
                "module": "03_video_content",
                "brand_name": brand_name,
                "sector": sector,
                "scripts_generated": 3,
                "videos_generated": 3,
                "teaser_style": teaser_style,
                "thumbnail_type": thumbnail_type,
                "recommended_voice": voice["id"],
                "llm_enhanced": bool(self._enhance_prompt("test", {}) != "test")
            }
        }
    
    def _personalize_script(self, text: str, **kwargs) -> str:
        """Personnalise un script avec les données de la marque"""
        replacements = {
            "{product}": kwargs.get("product_name", "notre produit"),
            "{brand}": kwargs.get("brand_name", "notre marque"),
            "{material}": kwargs.get("material", "matériaux d'exception"),
            "{detail}": kwargs.get("detail", "finitions soignées"),
            "{year}": kwargs.get("year", "1923"),
            "{code}": kwargs.get("code", "BIENVENUE"),
            "{duration}": kwargs.get("duration_days", "7"),
            "{problem}": kwargs.get("problem", "problèmes"),
            "{solution}": kwargs.get("solution", "solution"),
            "{benefit}": kwargs.get("benefits", ["qualité"])[0] if kwargs.get("benefits") else "qualité",
            "{result}": kwargs.get("result", "radieuse"),
            "{ingredient}": kwargs.get("ingredient", "ingrédients naturels"),
            "{feature}": kwargs.get("feature", "technologie avancée"),
            "{calories}": kwargs.get("calories", "500"),
            "{kg}": kwargs.get("kg", "10"),
            "{days}": kwargs.get("days", "30"),
            "{years}": kwargs.get("years", "5"),
            "{fit}": kwargs.get("fit", "parfaite"),
            "{color}": kwargs.get("color", "intemporelle"),
            "{audience}": kwargs.get("target_audience", "femmes élégantes"),
            "{influencers}": kwargs.get("influencers", "100+")
        }
        
        for placeholder, value in replacements.items():
            text = text.replace(placeholder, str(value))
        
        return text
    
    def _generate_video_prompt(self, duration: str, format: str, product_name: str, sector: str, script: str) -> str:
        """Génère un prompt pour la création vidéo"""
        aspect = "9:16 (1080x1920)" if format == "vertical" else "16:9 (1920x1080)"
        
        return f"""
Crée une vidéo publicitaire pour {product_name}:
- Durée: {duration}
- Format: {aspect}
- Secteur: {sector}
- Style: professionnel, adapté au secteur

Script:
{script}

Instructions:
- Utiliser des visuels lifestyle et produit
- Transitions dynamiques
- Musique adaptée au rythme du script
- Texte overlay sur les moments clés
- Logo de la marque en fin de vidéo
"""
    
    def _customize_teaser_prompt(self, prompt: str, product_name: str, brand_name: str, sector: str) -> str:
        """Personnalise le prompt du teaser"""
        return f"""
{prompt}

Produit: {product_name}
Marque: {brand_name}
Secteur: {sector}

Adaptez le teaser à l'univers de la marque.
"""
    
    def _customize_thumbnail_prompt(self, prompt: str, product_name: str, brand_name: str) -> str:
        """Personnalise le prompt de la miniature"""
        return f"""
{prompt}

Produit: {product_name}
Marque: {brand_name}

Adaptez les couleurs et le style à l'identité de la marque.
"""
    
    def _get_thumbnail_type(self, sector: str) -> str:
        """Détermine le type de miniature selon le secteur"""
        mapping = {
            "bijou": "product_focus",
            "luxe": "product_focus",

    def _get_teaser_style(self, sector: str) -> str:
