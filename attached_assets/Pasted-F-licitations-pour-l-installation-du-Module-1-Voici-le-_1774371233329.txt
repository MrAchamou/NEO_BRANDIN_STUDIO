```python
def get_product_photo_prompt(product_type: str, angle: str) -> str:
    """Génère un prompt pour photo produit selon le type et l'angle"""
    prompts = PRODUCT_PHOTO_PROMPTS.get(product_type, PRODUCT_PHOTO_PROMPTS["bijou"])
    return prompts.get(angle, prompts["front"])

def get_lifestyle_prompt(product_type: str, format_type: str) -> str:
    """Génère un prompt pour photo lifestyle selon le produit et le format"""
    prompts = LIFESTYLE_PROMPTS.get(product_type, LIFESTYLE_PROMPTS["bijou"])
    return prompts.get(format_type, prompts["feed"])

def get_detail_prompt(material: str) -> str:
    """Génère un prompt pour photo détail selon le matériau"""
    return DETAIL_PROMPTS.get(material, DETAIL_PROMPTS["métal"])

def get_before_after_prompt(scenario: str) -> dict:
    """Génère les prompts before/after pour un scénario donné"""
    return BEFORE_AFTER_PROMPTS.get(scenario, BEFORE_AFTER_PROMPTS["skin"])

def get_tryon_prompt(category: str, model_index: int) -> str:
    """Génère un prompt pour virtual try-on selon la catégorie"""
    prompts = VIRTUAL_TRYON_PROMPTS.get(category, VIRTUAL_TRYON_PROMPTS["vêtement"])
    key = f"model_{model_index}"
    return prompts.get(key, prompts["model_1"])

def get_carousel_prompt(style: str, slide: int) -> str:
    """Génère un prompt pour une slide de carrousel"""
    prompts = CAROUSEL_PROMPTS.get(style, CAROUSEL_PROMPTS["probleme_solution"])
    return prompts.get(f"slide_{slide}", prompts["slide_1"])
```

```python
from typing import Dict, Any
from modules.base_module import BaseModule
from modules.module_02_visual_content.prompts import (
    get_product_photo_prompt,
    get_lifestyle_prompt,
    get_detail_prompt,
    get_before_after_prompt,
    get_tryon_prompt,
    get_carousel_prompt
)

class VisualContentOrchestrator(BaseModule):
    """
    MODULE 02 — VISUAL CONTENT
    Génère tous les prompts pour créer le contenu visuel d'une marque
    """
    
    def __init__(self):
        super().__init__("VisualContent")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 02
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "product_type": "bijou",  # bijou, vêtement, sac, cosmétique, tech, fitness
            "product_name": "Montre Élégance",
            "product_colors": ["or rose", "noir"],
            "product_materials": ["acier inoxydable", "verre saphir"],
            "target_audience": "femmes_25_45",
            "brand_identity": {}  # du Module 01 (optionnel, pour contexte)
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_type = brief.get("product_type", "bijou")
        product_name = brief.get("product_name", "notre produit")
        product_colors = brief.get("product_colors", [])
        product_materials = brief.get("product_materials", [])
        target_audience = brief.get("target_audience", "mixte")
        
        # Déterminer les angles selon le type de produit
        angles = self._get_angles_for_product_type(product_type)
        
        # Déterminer les formats lifestyle selon le secteur
        formats = ["feed", "pinterest", "story"]
        
        # Déterminer les matériaux pour les photos détail
        materials = product_materials if product_materials else ["métal"]
        
        # Déterminer le scénario before/after selon le secteur
        scenario = self._get_before_after_scenario(sector)
        
        # Déterminer la catégorie try-on
        tryon_category = self._get_tryon_category(product_type)
        
        # Déterminer le style carrousel
        carousel_style = self._get_carousel_style(sector)
        
        self._log(f"Génération des prompts visuels pour {brand_name} ({product_type})")
        
        # 1. Photos produit (5 angles)
        product_photos = {}
        for angle in angles:
            base_prompt = get_product_photo_prompt(product_type, angle)
            custom_prompt = self._customize_product_prompt(
                base_prompt, product_name, product_colors, product_materials
            )
            enhanced = self._enhance_prompt(custom_prompt, {
                "brand": brand_name,
                "product": product_name,
                "type": product_type
            })
            product_photos[angle] = enhanced
        
        # 2. Photos lifestyle (3 formats)
        lifestyle_photos = {}
        for fmt in formats:
            base_prompt = get_lifestyle_prompt(product_type, fmt)
            custom_prompt = self._customize_lifestyle_prompt(
                base_prompt, product_name, sector, target_audience
            )
            enhanced = self._enhance_prompt(custom_prompt, {
                "brand": brand_name,
                "product": product_name,
                "format": fmt
            })
            lifestyle_photos[fmt] = enhanced
        
        # 3. Photos détail/texture (2 matériaux)
        detail_photos = {}
        for material in materials[:2]:
            base_prompt = get_detail_prompt(material)
            custom_prompt = self._customize_detail_prompt(base_prompt, material, product_name)
            enhanced = self._enhance_prompt(custom_prompt, {
                "material": material,
                "product": product_name
            })
            detail_photos[material] = enhanced
        
        # 4. Before/After
        before_after = get_before_after_prompt(scenario)
        before_after_enhanced = {
            "before": self._enhance_prompt(before_after["before"], {"scenario": scenario}),
            "after": self._enhance_prompt(before_after["after"], {"scenario": scenario})
        }
        
        # 5. Virtual Try-On (2 modèles)
        tryon_photos = {}
        for i in range(1, 3):
            base_prompt = get_tryon_prompt(tryon_category, i)
            custom_prompt = self._customize_tryon_prompt(
                base_prompt, product_name, product_type, target_audience
            )
            enhanced = self._enhance_prompt(custom_prompt, {
                "product": product_name,
                "model": i
            })
            tryon_photos[f"model_{i}"] = enhanced
        
        # 6. Carousel (5 slides)
        carousel = {}
        for slide in range(1, 6):
            base_prompt = get_carousel_prompt(carousel_style, slide)
            custom_prompt = self._customize_carousel_prompt(
                base_prompt, product_name, brand_name, sector, slide
            )
            enhanced = self._enhance_prompt(custom_prompt, {
                "product": product_name,
                "slide": slide,
                "style": carousel_style
            })
            carousel[f"slide_{slide}"] = enhanced
        
        # Retourner tous les prompts
        return {
            "visual_content": {
                "product_photos": {
                    "agent": "Product Display Agent / Image Generation",
                    "prompts": product_photos,
                    "parameters": {
                        "angles": angles,
                        "dimensions": "2000x2000",
                        "background": "white"
                    }
                },
                "lifestyle_photos": {
                    "agent": "Image Generation / Background Change",
                    "prompts": lifestyle_photos,
                    "parameters": {
                        "formats": ["feed (1:1)", "pinterest (4:5)", "story (9:16)"],
                        "dimensions": ["1080x1080", "1080x1350", "1080x1920"]
                    }
                },
                "detail_photos": {
                    "agent": "Image Generation",
                    "prompts": detail_photos,
                    "parameters": {
                        "materials": materials[:2],
                        "dimensions": "2000x2000"
                    }
                },
                "before_after": {
                    "agent": "Portrait Retouching Agent",
                    "prompts": before_after_enhanced,
                    "parameters": {
                        "scenario": scenario,
                        "format": "side_by_side",
                        "dimensions": "1080x1080"
                    }
                },
                "virtual_tryon": {
                    "agent": "AI Wardrobe / Virtual Try-on",
                    "prompts": tryon_photos,
                    "parameters": {
                        "models": 2,
                        "poses": ["front", "three_quarter"],
                        "dimensions": "1080x1350"
                    }
                },
                "carousel": {
                    "agent": "AI Poster Agent",
                    "prompts": carousel,
                    "parameters": {
                        "slides": 5,
                        "style": carousel_style,
                        "dimensions": "1080x1080"
                    }
                }
            },
            "metadata": {
                "module": "02_visual_content",
                "brand_name": brand_name,
                "product_type": product_type,
                "angles_generated": len(angles),
                "lifestyle_formats": len(formats),
                "tryon_models": 2,
                "carousel_slides": 5,
                "llm_enhanced": bool(self._enhance_prompt("test", {}) != "test")
            }
        }
    
    def _get_angles_for_product_type(self, product_type: str) -> list:

        angles_map = {

        mapping = {

            "tech": "education",

            "streetwear": "storytelling"
        }
        return mapping.get(sector, "probleme_solution")
    
    def _customize_product_prompt(self, prompt: str, product_name: str, colors: list, materials: list) -> str:
        """Personnalise le prompt photo produit"""
        custom = f"{prompt}\n\nProduit: {product_name}"
        if colors:
            custom += f"\nCouleurs: {', '.join(colors)}"
        if materials:
            custom += f"\nMatériaux: {', '.join(materials)}"
        return custom
    
    def _customize_lifestyle_prompt(self, prompt: str, product_name: str, sector: str, audience: str) -> str:
        """Personnalise le prompt photo lifestyle"""
        return f"{prompt}\n\nProduit: {product_name}\nSecteur: {sector}\nCible: {audience}"
    
    def _customize_detail_prompt(self, prompt: str, material: str, product_name: str) -> str:
        """Personnalise le prompt photo détail"""
        return f"{prompt}\n\nProduit: {product_name}\nMatériau: {material}"
    
    def _customize_tryon_prompt(self, prompt: str, product_name: str, product_type: str, audience: str) -> str:
        """Personnalise le prompt virtual try-on"""
        return f"{prompt}\n\nProduit: {product_name}\nType: {product_type}\nCible: {audience}"
    
    def _customize_carousel_prompt(self, prompt: str, product_name: str, brand_name: str, sector: str, slide: int) -> str:
        """Personnalise le prompt carrousel"""
        return f"{prompt}\n\nMarque: {brand_name}\nProduit: {product_name}\nSecteur: {sector}\nSlide: {slide}/5"
```
