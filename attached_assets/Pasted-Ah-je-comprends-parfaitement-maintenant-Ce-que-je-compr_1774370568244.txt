```
INPUT (brief client)

[LOGICIEL — Replit AI]
    ↓
OUTPUT (fichier texte)
    ↓
Contient des PROMPTS PARFAITS
    ↓
MOI (humain) → je copie/colle ces prompts dans RoboNeo.com
    ↓
RoboNeo génère les assets

Le logiciel est un **générateur de prompts**, pas un orchestrateur d'API. Les APIs à connecter sont **Claude, Gemini, OpenAI** — pour améliorer les prompts si besoin.

## 4. FICHIER utils/llm_client.py

```python
import os
from typing import Optional
from config import Config

class LLMClient:
    """Client pour améliorer les prompts avec Claude/Gemini/OpenAI"""
    
    def __init__(self):
        self.enabled = Config.ENABLE_LLM_ENHANCEMENT
        self.client = None
        self.provider = None
        
        if self.enabled:
            self._init_client()
    
    def _init_client(self):
        if Config.ANTHROPIC_API_KEY:
            try:
                import anthropic
                self.client = anthropic.Anthropic(api_key=Config.ANTHROPIC_API_KEY)
                self.provider = "claude"
            except ImportError:
                pass
        
        elif Config.GEMINI_API_KEY and not self.client:
            try:
                import google.generativeai as genai
                genai.configure(api_key=Config.GEMINI_API_KEY)
                self.client = genai.GenerativeModel('gemini-pro')
                self.provider = "gemini"
            except ImportError:
                pass
        
        elif Config.OPENAI_API_KEY and not self.client:
            try:
                from openai import OpenAI
                self.client = OpenAI(api_key=Config.OPENAI_API_KEY)
                self.provider = "openai"
            except ImportError:
                pass
    
    def enhance_prompt(self, base_prompt: str, context: dict) -> str:
        """Améliore un prompt en utilisant un LLM"""
        if not self.enabled or not self.client:
            return base_prompt
        
        system = f"""
        Tu es un expert en prompts pour RoboNeo.com, une plateforme de génération d'assets créatifs par IA.
        Améliore le prompt ci-dessous pour qu'il soit plus précis, détaillé et efficace.
        Garde la même structure mais ajoute des détails pertinents.
        Ne modifie pas l'intention principale. Retourne uniquement le prompt amélioré.
        """
        
        user = f"""
        Contexte:
        - Marque: {context.get('brand_name', 'inconnue')}
        - Secteur: {context.get('sector', 'general')}
        - Ton: {context.get('tone', 'professionnel')}
        - Valeurs: {', '.join(context.get('values', []))}
        
        Prompt original:
        {base_prompt}
        
        Prompt amélioré:
        """
        
        try:
            if self.provider == "claude":
                response = self.client.messages.create(
                    model="claude-3-sonnet-20240229",
                    max_tokens=1000,
                    system=system,
                    messages=[{"role": "user", "content": user}]
                )
                return response.content[0].text.strip()
            
            elif self.provider == "gemini":
                response = self.client.generate_content(f"{system}\n\n{user}")
                return response.text.strip()
            
            elif self.provider == "openai":
                response = self.client.chat.completions.create(
                    model="gpt-4-turbo-preview",
                    messages=[
                        {"role": "system", "content": system},
                        {"role": "user", "content": user}
                    ],
                    max_tokens=1000
                )
                return response.choices[0].message.content.strip()
        
        except Exception as e:
            print(f"Erreur LLM: {e}")
            return base_prompt
        
        return base_prompt

llm = LLMClient()
```

## 5. FICHIER utils/file_exporter.py

```python
import json
import os
from typing import Dict, Any
from datetime import datetime

class FileExporter:
    """Exporte les prompts générés dans un fichier"""
    
    def __init__(self, output_path: str = "output/prompts_roboneo.json"):
        self.output_path = output_path
        os.makedirs(os.path.dirname(output_path), exist_ok=True)
    
    def export(self, data: Dict[str, Any]) -> str:
        """Exporte les données en JSON"""
        # Ajouter métadonnées
        output = {
            "generated_at": datetime.now().isoformat(),
            "version": "1.0.0",
            "modules": data
        }
        
        with open(self.output_path, 'w', encoding='utf-8') as f:
            json.dump(output, f, ensure_ascii=False, indent=2)
        
        return self.output_path
    
    def export_txt(self, data: Dict[str, Any]) -> str:
        """Exporte les prompts en format texte lisible"""
        txt_path = self.output_path.replace('.json', '.txt')
        
        with open(txt_path, 'w', encoding='utf-8') as f:
            f.write("=" * 80 + "\n")
            f.write("PROMPTS POUR ROBONEO.COM\n")
            f.write(f"Généré le: {datetime.now().isoformat()}\n")
            f.write("=" * 80 + "\n\n")
            
            for module_name, module_data in data.items():
                f.write(f"\n{'=' * 80}\n")
                f.write(f"MODULE: {module_name}\n")
                f.write('=' * 80 + "\n\n")
                
                for submodule_name, submodule_data in module_data.items():
                    f.write(f"\n--- {submodule_name} ---\n\n")
                    if isinstance(submodule_data, dict):
                        for key, value in submodule_data.items():
                            f.write(f"{key}:\n{value}\n\n")
                    else:
                        f.write(f"{submodule_data}\n\n")
        
        return txt_path
```

## 6. FICHIER modules/base_module.py

```python
from abc import ABC, abstractmethod
from typing import Dict, Any
from utils.llm_client import llm
from utils.logger import logger

class BaseModule(ABC):
    """Classe de base pour tous les modules (01 à 10)"""
    
    def __init__(self, module_name: str):
        self.module_name = module_name
    
    @abstractmethod
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """Génère les prompts du module à partir du brief"""
        pass
    
    def _enhance_prompt(self, base_prompt: str, context: Dict) -> str:
        """Améliore un prompt avec LLM si activé"""
        return llm.enhance_prompt(base_prompt, context)
    
    def _log(self, message: str, level: str = "info"):
        print(f"[{self.module_name}] {message}")
```

class BrandIdentityOrchestrator(BaseModule):
    """
    MODULE 01 — BRAND IDENTITY
    Génère tous les prompts pour créer l'identité visuelle d'une marque
    """
    
    def __init__(self):
        super().__init__("BrandIdentity")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 01
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "tone": "luxe",
            "values": ["excellence", "prestige", "authenticité"],
            "style_pref": null  # optionnel, sinon auto-détection
        }
        """
        
        # Extraire les données du brief
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        tone = brief.get("tone", "professionnel")
        values = brief.get("values", ["qualité", "confiance"])
        style_pref = brief.get("style_pref", self._detect_style(sector))
        
        # Contexte pour amélioration des prompts
        context = {
            "brand_name": brand_name,
            "sector": sector,
            "tone": tone,
            "values": values,
            "style": style_pref
        }
        
        self._log(f"Génération des prompts pour {brand_name} ({sector})")
        
        # 1. Logo Generator
        logo_prompt = get_logo_prompt(style_pref, brand_name, sector, tone, values)

        # 2. Palette Generator

        # 4. Brand Guidelines
        guidelines_prompt = GUIDELINES_PROMPT
        guidelines_prompt_enhanced = self._enhance_prompt(guidelines_prompt, context)
        
        # Retourner tous les prompts
        return {
            "brand_identity": {
                "logo": {
                    "agent": "Brand Design Agent / Product Display Agent",

    }
    
    # Si des arguments sont passés en ligne de commande
    if len(sys.argv) > 1:
        try:
            with open(sys.argv[1], 'r', encoding='utf-8') as f:
                brief = json.load(f)
            print(f"Brief chargé depuis {sys.argv[1]}")
        except Exception as e:
            print(f"Erreur de chargement: {e}")
            print("Utilisation du brief par défaut")
    else:
        print("Aucun fichier brief fourni. Utilisation du brief de test.")
        print("Pour utiliser votre propre brief: python main.py mon_brief.json")
    
    print()
    print(f"Marque: {brief.get('brand_name')}")
    print(f"Secteur: {brief.get('sector')}")
    print(f"Ton: {brief.get('tone')}")
    print(f"Valeurs: {', '.join(brief.get('values', []))}")
    print()
    
    # Génération des prompts
    print("Génération des prompts en cours...")
    orchestrator = BrandIdentityOrchestrator()
    result = orchestrator.generate(brief)
    
    # Export
    exporter = FileExporter(Config.DEFAULT_OUTPUT_FILE)
    json_path = exporter.export(result)
    txt_path = exporter.export_txt(result)
    
    print()
    print("=" * 80)
    print("✅ GÉNÉRATION TERMINÉE")
    print("=" * 80)
    print()
    print("Fichiers générés:")
    print(f"  - {json_path}")
    print(f"  - {txt_path}")
    print()
    print("Pour utiliser les prompts:")
    print("1. Ouvrez le fichier .txt")
    print("2. Copiez chaque prompt dans RoboNeo.com")
    print("3. Utilisez l'agent indiqué pour chaque élément")

    print()

if __name__ == "__main__":
